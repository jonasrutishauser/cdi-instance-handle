25  public class ExtendedInstanceExtension implements Extension {
26      private Set<Annotation> qualifiers = new HashSet<>();
27  
28      void addProducer(@Observes BeforeBeanDiscovery event) {
29          event.addAnnotatedType(ExtendedInstanceProducer.class, ExtendedInstanceProducer.class.getName());
30      }
31  
32      void addQualifiers(@Observes ProcessInjectionPoint<?, ExtendedInstance<?>> event) {
33          qualifiers.addAll(event.getInjectionPoint().getQualifiers());
34      }
35  
36      void addAllQualifiers(@SuppressWarnings("rawtypes") @Observes ProcessBeanAttributes<ExtendedInstance> event) {
37          event.configureBeanAttributes().qualifiers(qualifiers);
38      }
39  
40      void setExtendedInstanceProducer(@Observes ProcessProducer<?, ExtendedInstance<?>> event, BeanManager beanManager) {
41          Producer<ExtendedInstance<?>> extendedInstanceProducer = event.getProducer();
42          event.configureProducer().produceWith(creationalContext -> {
43              InjectionPoint targetInjectionPoint = (InjectionPoint) beanManager.getInjectableReference(
44                      extendedInstanceProducer.getInjectionPoints().stream()
45                              .filter(ip -> InjectionPoint.class.equals(ip.getType())).findAny().get(),
46                      creationalContext);
47              InjectionPoint instanceInjectionPoint = extendedInstanceProducer.getInjectionPoints().stream()
48                      .filter(ip -> ip.getType() instanceof ParameterizedType).findAny().get();
49              Instance<?> instance = (Instance<?>) beanManager.getInjectableReference(new InjectionPoint() {
50                  @Override
51                  public Type getType() {
52                      if (targetInjectionPoint.getType() instanceof Class) {
53                          return new InstanceType(Object.class);
54                      }
55                      return new InstanceType(
56                              ((ParameterizedType) targetInjectionPoint.getType()).getActualTypeArguments()[0]);
57                  }
58  
59                  @Override
60                  public Set<Annotation> getQualifiers() {
61                      return targetInjectionPoint.getQualifiers();
62                  }
63  
64                  @Override
65                  public Bean<?> getBean() {
66                      return instanceInjectionPoint.getBean();
67                  }
68  
69                  @Override
70                  public Member getMember() {
71                      return instanceInjectionPoint.getMember();
72                  }
73  
74                  @Override
75                  public Annotated getAnnotated() {
76                      return instanceInjectionPoint.getAnnotated();
77                  }
78  
79                  @Override
80                  public boolean isDelegate() {
81                      return false;
82                  }
83  
84                  @Override
85                  public boolean isTransient() {
86                      return false;
87                  }
88              }, creationalContext);
89              return ExtendedInstanceProducer.create(beanManager, targetInjectionPoint, instance);
90          });
91      }
92  }
