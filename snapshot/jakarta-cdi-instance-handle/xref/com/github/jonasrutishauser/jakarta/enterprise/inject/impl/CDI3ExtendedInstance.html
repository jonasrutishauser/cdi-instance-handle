2   
3   import java.lang.annotation.Annotation;
4   import java.lang.reflect.Type;
5   import java.util.HashSet;
6   import java.util.Set;
7   
8   import com.github.jonasrutishauser.jakarta.enterprise.inject.ExtendedInstance;
9   
10  import jakarta.enterprise.inject.Instance;
11  import jakarta.enterprise.inject.spi.Bean;
12  import jakarta.enterprise.inject.spi.BeanManager;
13  import jakarta.enterprise.util.TypeLiteral;
14  
15  class CDI3ExtendedInstance<T> implements ExtendedInstance<T> {
16      
17      private final BeanManager beanManager;
18      private final Instance<T> instance;
19      private final Type targetType;
20      private final Set<Annotation> qualifiers;
21  
22      public CDI3ExtendedInstance(BeanManager beanManager, Instance<T> instance, Type targetType, Set<Annotation> qualifiers) {
23          this.beanManager = beanManager;
24          this.targetType = targetType;
25          this.instance = instance;
26          this.qualifiers = qualifiers;
27      }
28  
29      public ExtendedInstance<T> select(Annotation... qualifiers) {
30          return new CDI3ExtendedInstance<>(beanManager, instance.select(qualifiers), targetType, addedQualifiers(qualifiers));
31      }
32  
33      public <U extends T> ExtendedInstance<U> select(Class<U> subtype, Annotation... qualifiers) {
34          return new CDI3ExtendedInstance<>(beanManager, instance.select(subtype, qualifiers), subtype, addedQualifiers(qualifiers));
35      }
36  
37      public <U extends T> ExtendedInstance<U> select(TypeLiteral<U> subtype, Annotation... qualifiers) {
38          return new CDI3ExtendedInstance<>(beanManager, instance.select(subtype, qualifiers), subtype.getType(), addedQualifiers(qualifiers));
39      }
40  
41      @Override
42      public Instance<T> getInstance() {
43          return instance;
44      }
45  
46      @Override
47      public Handle<T> getPseudoScopeClosingHandle() {
48          return new HandleImpl();
49      }
50  
51      private Set<Annotation> addedQualifiers(Annotation... qualifiers) {
52          Set<Annotation> newQualifiers = new HashSet<>(this.qualifiers);
53          for (Annotation qualifier : qualifiers) {
54              newQualifiers.add(qualifier);
55          }
56          return newQualifiers;
57      }
58  
59      private class HandleImpl implements Handle<T> {
60          private T bean;
61  
62          public T get() {
63              if (bean == null) {
64                  bean = instance.get();
65              }
66              return bean;
67          }
68  
69          public Bean<T> getBean() {
70              return (Bean<T>) beanManager.resolve(beanManager.getBeans(targetType, qualifiers.toArray(Annotation[]::new)));
71          }
72  
73          public void destroy() {
74              if (bean != null) {
75                  instance.destroy(bean);
76                  bean = null;
77              }
78          }
79  
80          public void close() {
81              if (!beanManager.isNormalScope(getBean().getScope())) {
82                  destroy();
83              }
84          }
85      }
86  }
