3   import java.lang.annotation.Annotation;
4   import java.lang.reflect.Type;
5   import java.util.Collections;
6   import java.util.HashSet;
7   import java.util.Set;
8   
9   import com.github.jonasrutishauser.javax.enterprise.inject.ExtendedInstance;
10  
11  import javax.enterprise.inject.Instance;
12  import javax.enterprise.inject.spi.Bean;
13  import javax.enterprise.inject.spi.BeanManager;
14  import javax.enterprise.util.TypeLiteral;
15  
16  class CDIExtendedInstance<T> implements ExtendedInstance<T> {
17      
18      private final BeanManager beanManager;
19      private final Instance<T> instance;
20      private final Type targetType;
21      private final Set<Annotation> qualifiers;
22  
23      public CDIExtendedInstance(BeanManager beanManager, Instance<T> instance, Type targetType, Set<Annotation> qualifiers) {
24          this.beanManager = beanManager;
25          this.targetType = targetType;
26          this.instance = instance;
27          this.qualifiers = qualifiers;
28      }
29  
30      public ExtendedInstance<T> select(Annotation... qualifiers) {
31          return new CDIExtendedInstance<>(beanManager, instance.select(qualifiers), targetType, addedQualifiers(qualifiers));
32      }
33  
34      public <U extends T> ExtendedInstance<U> select(Class<U> subtype, Annotation... qualifiers) {
35          return new CDIExtendedInstance<>(beanManager, instance.select(subtype, qualifiers), subtype, addedQualifiers(qualifiers));
36      }
37  
38      public <U extends T> ExtendedInstance<U> select(TypeLiteral<U> subtype, Annotation... qualifiers) {
39          return new CDIExtendedInstance<>(beanManager, instance.select(subtype, qualifiers), subtype.getType(), addedQualifiers(qualifiers));
40      }
41  
42      @Override
43      public Instance<T> getInstance() {
44          return instance;
45      }
46  
47      @Override
48      public Handle<T> getPseudoScopeClosingHandle() {
49          return new HandleImpl();
50      }
51  
52      private Set<Annotation> addedQualifiers(Annotation... qualifiers) {
53          Set<Annotation> newQualifiers = new HashSet<>(this.qualifiers);
54          Collections.addAll(newQualifiers, qualifiers);
55          return newQualifiers;
56      }
57  
58      private class HandleImpl implements Handle<T> {
59          private T bean;
60  
61          public T get() {
62              if (bean == null) {
63                  bean = instance.get();
64              }
65              return bean;
66          }
67  
68          public Bean<T> getBean() {
69              return (Bean<T>) beanManager.resolve(beanManager.getBeans(targetType, qualifiers.toArray(new Annotation[0])));
70          }
71  
72          public void destroy() {
73              if (bean != null) {
74                  instance.destroy(bean);
75                  bean = null;
76              }
77          }
78  
79          public void close() {
80              if (!beanManager.isNormalScope(getBean().getScope())) {
81                  destroy();
82              }
83          }
84      }
85  }
